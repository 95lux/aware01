\subsubsection{Electrical Constraints}

\begin{table}[ht]
    \centering
    \label{tab:req-electrical}
    \begin{tabularx}{\linewidth}{@{} l X @{}}
        \toprule
        \textbf{ID} & \textbf{Constraint} \\
        \midrule
        \label{req:EC-01}EC-01 & All inputs must tolerate the full Eurorack signal range ($-10$\,V to $+10$\,V) without damage to downstream circuitry \\
        \label{req:EC-02}EC-02 & Gate inputs are protected via transistor clamping; CV inputs use rail-guarded opamp stages to prevent ADC overvoltage \\
        \label{req:EC-03}EC-03 & Audio inputs are attenuated, diode-protected, and AC-coupled to safeguard the codec analogue front end \\
        \label{req:EC-04}EC-04 & Power is derived from the Eurorack bus; a \qty{5}{V} buck converter supplies the WS2812 LEDs; a \qty{3.3}{V} LDO supplies the MCU and analogue circuitry \\
        \label{req:EC-05}EC-05 & PCB layout must minimise noise coupling between digital and analogue signal paths \\
        \label{req:EC-06}EC-06 & Mechanical dimensions conform to the Eurorack 3U standard per \textbf{IEC~60297} / \textbf{DIN~41494} \\
        \bottomrule
    \end{tabularx}
\end{table}
