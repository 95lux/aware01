	\section*{Project Plan - Hardware Rev2}
	
	\section{Hardware (Week 1)}
	\begin{itemize}[leftmargin=*]
		\item Order and assemble \textbf{Hardware Revision 2}.
		\item Verify correct assembly and power integrity.
	\end{itemize}
	
	\section{Functional Testing (Week 2--3)}
	\begin{itemize}[leftmargin=*]
		\item Validate core functionality of the module:
		\begin{itemize}
			\item \textbf{STM32 CubeMX}
			\begin{itemize}
				\item Configure pinout and peripherals in STM32 CubeMX.
				\item Set up project via CMake.
				\item Research and include available drivers and libraries:
				\begin{itemize}
					\item FreeRTOS
					\item WS2812 (optional)
					\item CMSIS DSP
					\item Custom Audio Codec Driver (to be implemented)
				\end{itemize}
			\end{itemize}
			\item \textbf{MCU}
			\begin{itemize}
				\item Ensure device is flashable and debuggable via SWD.
				\item Set up debugging environment and build system (CMake) in Visual Studio Code.
			\end{itemize}
			\item \textbf{Audio Codec}
			\begin{itemize}
				\item Confirm accessibility and configuration via I\textsuperscript{2}C.
				\item Test all audio inputs and outputs:
				\begin{itemize}
					\item Route input directly to output to verify circuitry and correct AD/DA conversion.
				\end{itemize}
				\item Evaluate signal integrity for suitability in DSP development.
			\end{itemize}
			\item \textbf{Interface}
			\begin{itemize}
				\item Confirm ADC readings from potentiometers.
				\item Test WS2812 PWM control.
			\end{itemize}
		\end{itemize}
	\end{itemize}
	
	\section{Audio Engine (Week 4)}
	\begin{itemize}[leftmargin=*]
		\item Implement usable I\textsuperscript{2}C drivers for the audio codec.
		\item Set up double-buffered audio I/O.
		\item Define and provide data structures for:
		\begin{itemize}
			\item DSP algorithm integration.
			\item Audio playback (recording/playback buffers).
		\end{itemize}
		\item Implement playback engine (variable pitch/playback speed).
		\item Set up FreeRTOS tasks:
		\begin{itemize}
			\item Audio task (and possible subtasks)
			\item Control Interface Task
			\item User Interface Task
		\end{itemize}
	\end{itemize}
	
	\section{DSP Development (Week 5--6)}
	\begin{itemize}[leftmargin=*]
		\item Implement real-time DSP algorithms (main module functionality, TBD).
		\item Initially, implement or use an IIR filter (available in CMSIS library).
		\item Implement basic granular synthesis functionality (time permitting).
	\end{itemize}
	
	\section{Control Interface (Week 7)}
	\begin{itemize}[leftmargin=*]
		\item Implement control inputs:
		\begin{itemize}
			\item CV inputs.
			\item V/Oct input with calibration (calibration and validation of tracking if time permits).
			\item Gate inputs.
		\end{itemize}
	\end{itemize}
	
	\section{User Interface (Week 8)}
	\begin{itemize}[leftmargin=*]
		\item Integrate and test faders (slide potentiometers).
		\item Implement RGB LED control.
		\item Handle button input with proper debouncing.
	\end{itemize}
